1. select ai.account_no, ai.customer_id, ai.account_type, bi.bank_name from account_info ai inner join bank_info bi on (ai.ifsc_code=bi.ifsc_code);
2. select ai.customer_id, ai.account_type, ai.account_no from account_info ai inner join bank_info bi on (ai.ifsc_code=bi.ifsc_code) where (bi.bank_name = 'HDFC') and ai.registration_date between '2012-01-12' and '2012-04-04';
3. select cpi.customer_id, cpi.customer_name, ai.account_no, ai.account_type, bi.bank_name from customer_personal_info cpi, account_info ai,bank_info bi where cpi.customer_id=ai.customer_id and ai.ifsc_code=bi.ifsc_code;
4. select customer_id, customer_name, gender, marital_status, concat(customer_name,'_',gender,'_',marital_status) as Unique_String from customer_personal_info;
5. select account_no, customer_id, registration_date, initial_deposit from account_info where initial_deposit between 15000 and 25000;
6. select customer_id, customer_name, date_of_birth, guardian_name from customer_personal_info where customer_name like 'J%';
7. select account_no, customer_id, concat(substring(customer_id from 3 for 3),substring(cast(account_no as varchar(16)) from 13 for 4)) as password from account_info;
8. select customer_id, customer_name, date_of_birth, marital_status, gender, guardian_name, contact_no, mail_id from customer_personal_info where gender='M' and marital_status='MARRIED';
9. select cpi.customer_id, cpi.customer_name, cpi. guardian_name, r.reference_acc_name from customer_personal_info cpi, customer_reference_info r where r.relation='FRIEND' and r.customer_id=cpi.customer_id;
10.select customer_id, account_no, cast((interest*initial_deposit/100) as decimal(7, 2)) as Interest from account_info order by Interest;
11.select a.customer_id, a.account_no, a.activation_date, b.customer_name from account_info a, customer_personal_info b where cast(a.activation_date as varchar)=to_char(to_date('10-APR-2012', 'DD-MON-YYYY'), 'YYYY-MM-DD') and a.customer_id=b.customer_id;
12.select ai.account_no, ai.customer_id, cpi.customer_name, bi.bank_name, bi.branch_name, bi.ifsc_code, cpi.citizenship, cast((ai.interest*ai.initial_deposit/100) as decimal(7, 2)) as Interest, ai.initial_deposit from account_info ai, bank_info bi, customer_personal_info cpi where ai.ifsc_code=bi.ifsc_code and ai.customer_id=cpi.customer_id;
13.select cpi.customer_id, cpi.customer_name, cpi.date_of_birth, cpi.guardian_name, cpi.contact_no,cpi.mail_id, r. reference_acc_name from customer_personal_info cpi, customer_reference_info r where r.customer_id=cpi.customer_id and cpi.identification_doc_type='PASSPORT';
14.select cpi.customer_id, cpi.customer_name, ai.account_no, ai.account_type, ai.initial_deposit, cast((interest*initial_deposit/100) as decimal(7, 2)) as Interest from account_info ai, customer_personal_info cpi order by Interest desc limit 1;
15.select cpi.customer_id, cpi.customer_name, ai.account_no, ai.account_type, ai.initial_deposit, cast((interest*initial_deposit/100) as decimal(7, 2)) as Interest from account_info ai, customer_personal_info cpi order by ai.interest desc limit 1;
16.select cpi.customer_id, cpi.customer_name, ai.account_no, bi.bank_name, cpi.contact_no, cpi.mail_id from customer_personal_info cpi, account_info ai, bank_info bi where ai.ifsc_code=bi.ifsc_code and cpi.customer_id=ai.customer_id and right(cpi.address, 9)='BANGALORE';
17.select ai.customer_id, bi.bank_name, bi.branch_name, bi.ifsc_code, ai.registration_date, ai.activation_date from account_info ai, bank_info bi where ai.ifsc_code=bi.ifsc_code and extract('month' from ai.activation_date)=03;
18.select cpi.customer_id, cpi.customer_name, ai.account_no, ai.account_type, ai.interest, ai.initial_deposit, cast((interest*initial_deposit/100) as decimal(7, 2)) as InterestAmount from account_info ai, customer_personal_info cpi where cpi.customer_id=ai.customer_id;
19.select cpi.customer_id, cpi.customer_name,cpi.date_of_birth, cpi.guardian_name, cpi.contact_no, r.reference_acc_name from customer_personal_info cpi, customer_reference_info r where r.customer_id=cpi.customer_id and r.reference_acc_name='RAGHUL';
20.select customer_id, customer_name, concat('+91-',substring(cast(contact_no as varchar),1,3),'-',substring(cast(contact_no as varchar),4,3),'-',substring(cast(contact_no as varchar), 7)) as Phone_No from customer_personal_info;
21.select ai.account_no, ai.account_type, cpi.customer_id, cpi.customer_name, cpi.date_of_birth, cpi.guardian_name, cpi.contact_no, cpi.mail_id, cpi.gender, r.reference_acc_name, r.reference_acc_no, ai.registration_date, ai.activation_date, ai.activation_date-ai.registration_date as noofactivedays, bi.bank_name, bi.branch_name, ai.initial_deposit from customer_personal_info cpi, account_info ai, bank_info bi, customer_reference_info r where cpi.customer_id=ai.customer_id and cpi.customer_id=r.customer_id and ai.ifsc_code=bi.ifsc_code;
22.select cpi.customer_id, cpi.customer_name, cpi.guardian_name, cpi.identification_doc_type, r.reference_acc_name, ai.account_type, ai.ifsc_code, bi.bank_name, ((date_part('year', current_date)-date_part('year', ai.activation_date))*ai.interest*ai.initial_deposit/100)+ai.initial_deposit as current_balance from customer_personal_info cpi, account_info ai, bank_info bi, customer_reference_info r where cpi.customer_id=ai.customer_id and r.customer_id=cpi.customer_id and bi.ifsc_code=ai.ifsc_code and ai.account_type='SAVINGS';
23.select cpi.customer_id, cpi.customer_name, ai.accountb_no, ai.account_type, ai.interest, ai.initial_deposit, case initial_deposit when 20000 then 'High' when 16000 then 'Moderate' when 10000 then 'Average' when 5000 then 'Low' else 'very low' end as deposit_status from customer_personal_info cpi, account_info ai where ai.customer_id=cpi.customer_id;
